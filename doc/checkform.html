Stip

CheckForm 是基于Stip的表单验证提示插件。

如何使用

在页面引入 stip.css, stip.js, checkform.js 3个文件即可

<link rel="stylesheet" href="css/Stip.css" />
<script src="js/Stip.js"></script>
<script src="js/checkform.js"></script>

验证一个input

okay 方法返回一个布尔值，表示当前是否验证通过全部规则

var checkfrom = [ //配置验证规则
	{
		id:"testinput", // input id
		type:{rule:"int", msg:"必须为数字"}, //类型必须是int 错误提示msg
		minValue:{rule:2, msg:"最小值为2"}, //最小值 2 错误提示msg
		maxValue:{rule:10}//最大值 错误提示 checkform自带
	}
];
	
var check = new CheckForm(checkfrom);//实例化

$('#test').click(function(){
	alert( check.okay() );// okay() 立即验证所有规则，所有规则正确返回 true 否则false
});

获取值

使用getValues() 方法获取全部的值，也可以使用getValue()，传递一个ID 获取具体的某个值

var rule = [
	{
		id:"abced",
		maxLength:{rule:6,msg:"hello"}
	}
];
var reg = new CheckForm(rule);

var json = reg.getValues();//返回一个json数据格式，key是 id, 值是value
/*
	json = {
		"abced":val
	}
*/

var val = reg.getValue("abced");// 返回 abced 文本框的值

更多验证规则

验证规则不仅仅局限于 输入类型 最大 最小，请看下面

[
	{
		id:"test",
		type:{rule: 'int' | 'string' | 'float', msg:String}, //输入类型 
		minValue:{rule:number, msg:String}, //最小值
		maxValue:{rule:number, msg:String}, //最大值
		minLength:{rule:number, msg:String}, // 最小长度
		maxLength:{rule:number, msg:String}, //最大长度
		email:{rule:true, msg:String}, //必须为Email
		reg:{rule:RegExp, msg:String} //匹配一个正则表达式
	}
]
