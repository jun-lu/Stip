注册实例

下面模拟一个用户注册的场景使用 checkform,Stip来制作

-------html代码----------------------------------------------------------------------
<ul class="exa-list">
<li><label for="email">邮箱：</label><input type="text" id="email"></li>
<li><label for="passworld">密码：</label><input type="text" id="password"></li>
<li><label for="passworld2">确认密码：</label><input type="text" id="password2"></li>
<li><label for="userName">姓名：</label><input type="text" id="userName"></li>
<li><label for="reg"></label><input type="button" value="立即注册" id="reg"></li>
</ul>
-------js代码----------------------------------------------------------------------
var json = [
	{
		id:"email",
		email:{rule:true,msg:"请添加正确的邮箱格式"}

	},
	{
		id:"password",
		minLength:{rule:6,msg:"密码最少6位"},
		maxLength:{rule:16,msg:"密码最长16位"}

	},
	{
		id:"password2",
		minLength:{rule:6,msg:"确认密码"},
		maxLength:{rule:16,msg:"确认密码"}
	},
	{
		id:"userName",
		reg:{rule:/[a-zA-Z]{3,20}/,msg:"英文3-20个字符"}
	}
	
];

var reg = new  CheckForm(json);

$('#reg').click(function(){
	if( reg.okay() ){
		if(reg.getValue("password") == reg.getValue("password2")){
			//alert(reg.getValues()); 你可以在调试面板调用这个方法查看表单的所有值
			alert("注册成功");
		}else{
			alert("两次密码不一致");
		}
	};
});
